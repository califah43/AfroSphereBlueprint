// ADMIN COMMAND SYSTEM FOR AFROSPHERE

// Admin roles
const ADMIN_ROLES = {
  OWNER: "owner",
  SUPER_ADMIN: "super_admin",
  MODERATOR: "moderator",
};

// Admin permissions
const ADMIN_PERMISSIONS = {
  BAN_USER: "ban_user",
  SUSPEND_USER: "suspend_user",
  DISABLE_USER: "disable_user",
  VERIFY_USER: "verify_user",
  REMOVE_POST: "remove_post",
  VIEW_REPORTS: "view_reports",
  CREATE_BADGE: "create_badge",
  ASSIGN_BADGE: "assign_badge",
  REMOVE_BADGE: "remove_badge",
  CREATE_ADMIN: "create_admin",
  REMOVE_ADMIN: "remove_admin",
};

// Admin Command Handler
async function executeAdminCommand(adminId, command, payload) {
  const admin = await db.admins.get(adminId);
  if (!admin) throw "INVALID_ADMIN";

  if (!admin.permissions.includes(command))
    throw "NOT_ALLOWED";

  switch (command) {
    case ADMIN_PERMISSIONS.BAN_USER:
      return banUser(payload.userId);

    case ADMIN_PERMISSIONS.SUSPEND_USER:
      return suspendUser(payload.userId, payload.duration);

    case ADMIN_PERMISSIONS.DISABLE_USER:
      return disableUser(payload.userId);

    case ADMIN_PERMISSIONS.REMOVE_POST:
      return deletePost(payload.postId);

    case ADMIN_PERMISSIONS.VIEW_REPORTS:
      return fetchReports();

    case ADMIN_PERMISSIONS.VERIFY_USER:
      return verifyUser(payload.userId);

    case ADMIN_PERMISSIONS.CREATE_BADGE:
      return createBadge(payload.badgeData);

    case ADMIN_PERMISSIONS.ASSIGN_BADGE:
      return assignBadgeToUser(payload.userId, payload.badgeId);

    case ADMIN_PERMISSIONS.REMOVE_BADGE:
      return removeBadgeFromUser(payload.userId, payload.badgeId);

    case ADMIN_PERMISSIONS.CREATE_ADMIN:
      return createNewAdmin(payload.email, payload.role);

    case ADMIN_PERMISSIONS.REMOVE_ADMIN:
      return removeAdmin(payload.adminId);

    default:
      throw "UNKNOWN_COMMAND";
  }
}

// COMMAND FUNCTIONS
async function banUser(userId) {
  await db.users.update(userId, {
    status: "banned",
    bannedAt: Date.now(),
  });
}

async function suspendUser(userId, duration) {
  await db.users.update(userId, {
    status: "suspended",
    suspendedUntil: Date.now() + duration,
  });
}

async function disableUser(userId) {
  await db.users.update(userId, {
    status: "disabled",
  });
}

async function verifyUser(userId) {
  await db.users.update(userId, {
    verified: true,
  });
}

async function deletePost(postId) {
  await db.posts.delete(postId);
}

async function fetchReports() {
  return await db.reports.all();
}

async function createBadge(badgeData) {
  return await db.badges.create(badgeData);
}

async function assignBadgeToUser(userId, badgeId) {
  return await db.users.update(userId, {
    badges: db.arrayUnion(badgeId),
  });
}

async function removeBadgeFromUser(userId, badgeId) {
  return await db.users.update(userId, {
    badges: db.arrayRemove(badgeId),
  });
}

async function createNewAdmin(email, role) {
  return await db.admins.create({
    email,
    role,
    permissions: getPermissionsForRole(role),
  });
}

async function removeAdmin(adminId) {
  return await db.admins.delete(adminId);
}

function getPermissionsForRole(role) {
  switch (role) {
    case ADMIN_ROLES.OWNER:
      return Object.values(ADMIN_PERMISSIONS);

    case ADMIN_ROLES.SUPER_ADMIN:
      return [
        ADMIN_PERMISSIONS.BAN_USER,
        ADMIN_PERMISSIONS.SUSPEND_USER,
        ADMIN_PERMISSIONS.DISABLE_USER,
        ADMIN_PERMISSIONS.REMOVE_POST,
        ADMIN_PERMISSIONS.VERIFY_USER,
        ADMIN_PERMISSIONS.ASSIGN_BADGE,
        ADMIN_PERMISSIONS.VIEW_REPORTS,
      ];

    case ADMIN_ROLES.MODERATOR:
      return [
        ADMIN_PERMISSIONS.REMOVE_POST,
        ADMIN_PERMISSIONS.VIEW_REPORTS,
      ];

    default:
      return [];
  }
}
