2) RELIABLE LIKES / COMMENTS COUNTS (Firestore + Cloud Functions + Client)

Goal: counts never go wrong, no lost updates, consistent counts for posts, comments, and reactions.

Data model (Firestore)

Collections:

/users/{uid}
 /posts/{postId}         // doc contains likesCount, commentsCount
 /posts/{postId}/likes/{likeId}    // each like doc { uid, createdAt }
 /posts/{postId}/comments/{commentId} // each comment doc { uid, text, createdAt, likesCount }
 /posts/{postId}/comments/{commentId}/likes/{likeId}

Why this model
	•	likes and comments are subcollections for easy listing.
	•	likesCount and commentsCount are materialized counters on the post doc updated atomically via Cloud Function transaction so clients can render counts fast.
	•	For per-comment likes we do the same (a likesCount field on the comment doc).

⸻

Cloud Function (Node.js) — increment/decrement counters atomically

Place in functions/index.js (Firebase Cloud Functions):

// functions/index.js
const functions = require('firebase-functions');
const admin = require('firebase-admin');
admin.initializeApp();

const db = admin.firestore();

// When a like doc is created under posts/{postId}/likes/{likeId}
exports.onPostLikeCreate = functions.firestore
  .document('posts/{postId}/likes/{likeId}')
  .onCreate(async (snap, context) => {
    const { postId } = context.params;
    const postRef = db.collection('posts').doc(postId);

    await db.runTransaction(async t => {
      const postDoc = await t.get(postRef);
      const prev = postDoc.exists && postDoc.data().likesCount ? postDoc.data().likesCount : 0;
      t.update(postRef, { likesCount: prev + 1 });
    });
  });

// When a like doc deleted
exports.onPostLikeDelete = functions.firestore
  .document('posts/{postId}/likes/{likeId}')
  .onDelete(async (snap, context) => {
    const { postId } = context.params;
    const postRef = db.collection('posts').doc(postId);

    await db.runTransaction(async t => {
      const postDoc = await t.get(postRef);
      const prev = postDoc.exists && postDoc.data().likesCount ? postDoc.data().likesCount : 0;
      const next = Math.max(0, prev - 1);
      t.update(postRef, { likesCount: next });
    });
  });

// Comment created: increment commentsCount on post
exports.onCommentCreate = functions.firestore
  .document('posts/{postId}/comments/{commentId}')
  .onCreate(async (snap, context) => {
    const { postId } = context.params;
    const postRef = db.collection('posts').doc(postId);
    await db.runTransaction(async t => {
      const postDoc = await t.get(postRef);
      const prev = postDoc.exists && postDoc.data().commentsCount ? postDoc.data().commentsCount : 0;
      t.update(postRef, { commentsCount: prev + 1 });
    });
  });

exports.onCommentDelete = functions.firestore
  .document('posts/{postId}/comments/{commentId}')
  .onDelete(async (snap, context) => {
    const { postId } = context.params;
    const postRef = db.collection('posts').doc(postId);
    await db.runTransaction(async t => {
      const postDoc = await t.get(postRef);
      const prev = postDoc.exists && postDoc.data().commentsCount ? postDoc.data().commentsCount : 0;

  });

// Similarly for comment likes:
exports.onCommentLikeCreate = functions.firestore
  .document('posts/{postId}/comments/{commentId}/likes/{likeId}')
  .onCreate(async (snap, context) => {
    const { postId, commentId } = context.params;
    const commentRef = db.collection('posts').doc(postId).collection('comments').doc(commentId);
    await db.runTransaction(async t => {
      const c = await t.get(commentRef);
      const prev = c.exists && c.data().likesCount ? c.data().likesCount : 0;
      t.update(commentRef, { likesCount: prev + 1 });
    });
  });

exports.onCommentLikeDelete = functions.firestore
  .document('posts/{postId}/comments/{commentId}/likes/{likeId}')
  .onDelete(async (snap, context) => {
    const { postId, commentId } = context.params;
    const commentRef = db.collection('posts').doc(postId).collection('comments').doc(commentId);

      t.update(commentRef, { likesCount: Math.max(0, prev - 1) });
    });
  });

Notes:
	•	Transactions ensure counters do not get out of sync under concurrent writes.
	•	Deploy functions with firebase deploy --only functions.

Client-side examples

React Native (Firestore)

Add like:

// addLikeToPost.js
import firestore from '@react-native-firebase/firestore';

export async function toggleLikePost(postId, uid) {
  const likesRef = firestore().collection('posts').doc(postId).collection('likes');
  const likeDoc = likesRef.doc(uid); // use uid as likeId so user can only like once
  const doc = await likeDoc.get();
  if (doc.exists) {
    // unlike
    await likeDoc.delete();
  } else {
    await likeDoc.set({ uid, createdAt: firestore.FieldValue.serverTimestamp() });
  }
}

⸻

Common pitfalls & fixes
	•	Don’t store likes as an array on the post doc. Arrays cause concurrency problems and doc size limits.
	•	Use user UID as likeId so a user can only create one like doc per post — easy toggle.
	•	Always rely on Cloud Functions to update materialized counters (so clients only read likesCount and commentsCount).
	•	Rate limits: Cloud Functions can get many triggers — keep functions small and idempotent.

Flutter (use flutter_localizations + intl or easy_localization)

// main.dart
import 'package:flutter/material.dart';
import 'package:easy_localization/easy_localization.dart';

Future<void> main() async {
