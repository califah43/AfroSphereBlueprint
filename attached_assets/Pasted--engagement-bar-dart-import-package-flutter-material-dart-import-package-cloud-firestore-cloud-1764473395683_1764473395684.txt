// engagement_bar.dart
import 'package:flutter/material.dart';
import 'package:cloud_firestore/cloud_firestore.dart';
import 'package:share_plus/share_plus.dart';

class EngagementBar extends StatefulWidget {
  final String postId;
  final bool initiallyLiked;
  final int initialLikes;
  final int commentsCount;
  final int viewsCount;
  final bool initiallySaved;

  // callbacks optional
  final Future<void> Function(String postId)? onLikeBackend;
  final Future<void> Function(String postId)? onSaveBackend;

  const EngagementBar({
    Key? key,
    required this.postId,
    this.initiallyLiked = false,
    this.initialLikes = 0,
    this.commentsCount = 0,
    this.viewsCount = 0,
    this.initiallySaved = false,
    this.onLikeBackend,
    this.onSaveBackend,
  }) : super(key: key);

  @override
  State<EngagementBar> createState() => _EngagementBarState();
}

class _EngagementBarState extends State<EngagementBar>
    with SingleTickerProviderStateMixin {
  late bool isLiked;
  late bool isSaved;
  late int likes;
  late AnimationController _heartController;
  late Animation<double> _heartScale;

  @override
  void initState() {
    super.initState();
    isLiked = widget.initiallyLiked;
    isSaved = widget.initiallySaved;
    likes = widget.initialLikes;

    _heartController = AnimationController(
      vsync: this,
      duration: const Duration(milliseconds: 300),
    );
    _heartScale = Tween<double>(begin: 1.0, end: 1.25).animate(
      CurvedAnimation(parent: _heartController, curve: Curves.elasticOut),
    );
  }

  @override
  void dispose() {
    _heartController.dispose();
    super.dispose();
  }

  Future<void> _toggleLike() async {
    setState(() {
      isLiked = !isLiked;
      likes += isLiked ? 1 : -1;
    });

    // local optimistic animation
    if (isLiked) {
      _heartController.forward().then((_) => _heartController.reverse());
    }

    // backend call (optional)
    try {
      if (widget.onLikeBackend != null) {
        await widget.onLikeBackend!(widget.postId);
      } else {
        // default Firestore toggle (see backend snippet below)
        await _toggleLikeFirestore(widget.postId, isLiked);
      }
    } catch (e) {
      // revert on error
      setState(() {
        isLiked = !isLiked;
        likes += isLiked ? 1 : -1;
      });
    }
  }

  Future<void> _toggleSave() async {
    setState(() => isSaved = !isSaved);
    try {
      if (widget.onSaveBackend != null) {
        await widget.onSaveBackend!(widget.postId);
      } else {
        await _toggleSaveFirestore(widget.postId, isSaved);
      }
    } catch (e) {
      setState(() => isSaved = !isSaved);
    }
  }

  void _openComments() {
    // navigate to comments screen — implement navigation in your app
    Navigator.of(context).pushNamed('/comments', arguments: widget.postId);
  }

  void _sharePost() {
    Share.share('Check this out 👇\nhttps://afrosphere.app/post/${widget.postId}');
  }

  @override
  Widget build(BuildContext context) {
    // Unique visual: soft pill with segmented icons + subtle glass effect
    return Container(
      margin: const EdgeInsets.symmetric(horizontal: 16, vertical: 10),
      padding: const EdgeInsets.symmetric(horizontal: 8, vertical: 6),
      decoration: BoxDecoration(
        color: Theme.of(context).brightness == Brightness.dark
            ? Colors.black.withOpacity(0.45)
            : Colors.white.withOpacity(0.9),
        borderRadius: BorderRadius.circular(999),
        boxShadow: [
          BoxShadow(
            color: Colors.black.withOpacity(0.12),
            blurRadius: 12,
            offset: const Offset(0, 6),
          )
        ],
      ),
      child: Row(
        mainAxisSize: MainAxisSize.max,
        children: [
          // Left: Like (animated)
          GestureDetector(
            onTap: _toggleLike,
            onLongPress: () {
              // optional: show reaction sheet (love, clap, fire, etc.)
              _showReactionSheet();
            },
            child: Row(
              children: [
                ScaleTransition(
                  scale: _heartScale,
                  child: Icon(
                    isLiked ? Icons.favorite : Icons.favorite_border,
                    color: isLiked ? Colors.redAccent : Colors.white,
                    size: 22,
                  ),
                ),
                const SizedBox(width: 8),
                Text(
                  _compactNumber(likes),
                  style: const TextStyle(fontWeight: FontWeight.w600),
                )
              ],
            ),
          ),

          const SizedBox(width: 18),

          // Comment
          InkWell(
            onTap: _openComments,
            borderRadius: BorderRadius.circular(20),
            child: Row(
              children: [
                const Icon(Icons.chat_bubble_outline, size: 20),
                const SizedBox(width: 8),
                Text(
                  _compactNumber(widget.commentsCount),
                  style: const TextStyle(fontWeight: FontWeight.w600),
                )
              ],
            ),
          ),

          const Spacer(),

          // View count (subtle)
          Row(
            children: [
              const Icon(Icons.remove_red_eye_outlined, size: 18),
              const SizedBox(width: 6),
              Text(
                _compactNumber(widget.viewsCount),
                style: const TextStyle(fontSize: 13, color: Colors.white70),
              ),
            ],
          ),

          const SizedBox(width: 16),

          // Save
          InkWell(
            onTap: _toggleSave,
            child: Icon(
              isSaved ? Icons.bookmark : Icons.bookmark_border,
              size: 20,
            ),
          ),

          const SizedBox(width: 12),

          // Share
          InkWell(
            onTap: _sharePost,
            child: const Icon(
              Icons.ios_share_outlined,
              size: 20,
            ),
          ),
        ],
      ),
    );
  }

  void _showReactionSheet() {
    showModalBottomSheet(
      context: context,
      backgroundColor: Colors.transparent,
      builder: (_) {
        return Container(
          padding: const EdgeInsets.symmetric(horizontal: 20, vertical: 14),
          margin: const EdgeInsets.all(20),
          decoration: BoxDecoration(
            color: Theme.of(context).canvasColor,
            borderRadius: BorderRadius.circular(16),
          ),
          child: Row(
            mainAxisAlignment: MainAxisAlignment.spaceEvenly,
            children: [
              _reactionIcon('🔥', () => _sendReaction('fire')),
              _reactionIcon('👏', () => _sendReaction('clap')),
              _reactionIcon('❤️', () => _sendReaction('love')),
              _reactionIcon('✨', () => _sendReaction('sparkle')),
            ],
          ),
        );
      },
    );
  }

  Widget _reactionIcon(String emoji, VoidCallback cb) => InkWell(
        onTap: () {
          Navigator.of(context).pop();
          cb();
        },
        child: Text(
          emoji,
          style: const TextStyle(fontSize: 30),
        ),
      );

    try {

  }

  String _compactNumber(int n) {
    if (n >= 1000000) return '${(n / 1000000).toStringAsFixed(1)}M';
    if (n >= 1000) return '${(n / 1000).toStringAsFixed(1)}K';
    return n.toString();
  }

  // default Firestore toggle (use Firestore instance in your app)
  Future<void> _toggleLikeFirestore(String postId, bool liked) async {
    final firestore = FirebaseFirestore.instance;
    final postRef = firestore.collection('posts').doc(postId);
    final likeRef = postRef.collection('likes').doc(/* currentUid */ 'CURRENT_UID');

    await firestore.runTransaction((tx) async {
      final snapshot = await tx.get(postRef);
      if (!snapshot.exists) return;

      final currentLikes = (snapshot.data()?['likesCount'] ?? 0) as int;
      final newLikes = liked ? currentLikes + 1 : (currentLikes - 1).clamp(0, 1 << 30);

      tx.update(postRef, {'likesCount': newLikes});
      if (liked) {
        tx.set(likeRef, {
          'uid': 'CURRENT_UID',
          'createdAt': FieldValue.serverTimestamp(),
        });
      } else {
        tx.delete(likeRef);
      }
    });
  }

  Future<void> _toggleSaveFirestore(String postId, bool saved) async {
    final firestore = FirebaseFirestore.instance;
    final userSavedRef = firestore
        .collection('users')
        .doc('CURRENT_UID')
        .collection('saved')
        .doc(postId);

    if (saved) {
      await userSavedRef.set({'postId': postId, 'createdAt': FieldValue.serverTimestamp()});
    } else {
      await userSavedRef.delete();
    }
  }
}
