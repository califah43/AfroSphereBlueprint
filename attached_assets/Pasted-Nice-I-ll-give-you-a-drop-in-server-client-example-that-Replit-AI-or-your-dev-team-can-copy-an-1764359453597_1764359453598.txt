3) index.js — Express server (create accounts + create posts)

// index.js
const express = require('express');
const admin = require('firebase-admin');
const multer = require('multer');
const cors = require('cors');
const { Readable } = require('stream');

const app = express();
app.use(cors());
app.use(express.json());

// ----- Initialize Firebase Admin using SERVICE_ACCOUNT_JSON env var -----
if (!process.env.SERVICE_ACCOUNT_JSON) {
  console.error('Missing SERVICE_ACCOUNT_JSON env var');
  process.exit(1);
}

const serviceAccount = JSON.parse(process.env.SERVICE_ACCOUNT_JSON);

admin.initializeApp({
  credential: admin.credential.cert(serviceAccount),
  storageBucket: process.env.STORAGE_BUCKET || `${serviceAccount.project_id}.appspot.com`
});

const db = admin.firestore();
const bucket = admin.storage().bucket();

// --------- Simple middleware to verify Firebase ID token -----------
async function verifyFirebaseIdToken(req, res, next) {
  const authHeader = req.headers.authorization || '';
  const match = authHeader.match(/Bearer (.+)/);
  if (!match) return res.status(401).json({ error: 'No Authorization token provided' });
  const idToken = match[1];

  try {
    const decoded = await admin.auth().verifyIdToken(idToken);
    req.uid = decoded.uid;
    next();
  } catch (err) {
    console.error('Error verifying ID token:', err);
    return res.status(401).json({ error: 'Invalid ID token' });
  }
}

// ------------------ Endpoint: Create Account (admin-created) ------------------
// NOTE: For normal client-side signups use Firebase client SDK (safer). This route
// is useful if you want server-side creation, tests, or seeded users.
app.post('/api/createAccount', async (req, res) => {
  try {
    const { email, password, displayName, username } = req.body;
    if (!email || !password || !username) {
      return res.status(400).json({ error: 'email, password, username required' });
    }

    // Create user in Firebase Auth
    const userRecord = await admin.auth().createUser({
      email,
      password,
      displayName,
    });

    // Create user doc in Firestore
    const userDoc = {
      uid: userRecord.uid,
      email: userRecord.email,
      displayName: displayName || '',
      username,
      avatar: null,
      bio: '',
      followersCount: 0,
      followingCount: 0,
      createdAt: admin.firestore.FieldValue.serverTimestamp()
    };

    await db.collection('users').doc(userRecord.uid).set(userDoc);

    return res.json({ success: true, uid: userRecord.uid, userDoc });
  } catch (err) {
    console.error('createAccount error:', err);
    // Friendly error message
    return res.status(500).json({ error: err.message || 'Server error' });
  }
});

// ------------------ Upload handling for posts ------------------
// multer memory storage (we upload file buffer directly to Firebase Storage)
const upload = multer({ storage: multer.memoryStorage(), limits: { fileSize: 50 * 1024 * 1024 } }); // 50MB limit

// ------------------ Endpoint: Create Post (authenticated) ------------------
app.post('/api/createPost', verifyFirebaseIdToken, upload.single('media'), async (req, res) => {
  try {
    const uid = req.uid;
    const { caption = '', tags = '' } = req.body; // tags as comma-separated
    const file = req.file; // optional if text-only posts allowed (here you required media)

    if (!file) {
      return res.status(400).json({ error: 'media file is required' });
    }

    // Generate a storage file path, e.g., posts/{uid}/{timestamp}_{originalname}
    const timestamp = Date.now();
    const ext = (file.originalname && file.originalname.split('.').pop()) || 'bin';
    const filename = `posts/${uid}/${timestamp}_${Math.random().toString(36).slice(2)}.${ext}`;
    const fileRef = bucket.file(filename);

    // Convert buffer to stream and upload
    const stream = Readable.from(file.buffer);
    const writeStream = fileRef.createWriteStream({
      metadata: {
        contentType: file.mimetype
      },
      resumable: false
    });

    stream.pipe(writeStream);

    await new Promise((resolve, reject) => {
      writeStream.on('finish', resolve);
      writeStream.on('error', reject);
    });

    // Make file public URL (optional). Better: use signed URLs or secure access tied to rules.
    // For simplicity, store the storage path and a downloadable URL.
    const publicUrl = `https://storage.googleapis.com/${bucket.name}/${filename}`;

    // Create the post document in Firestore
    const postRef = db.collection('posts').doc();
    const postData = {
      postId: postRef.id,
      uid,
      mediaUrl: publicUrl,
      storagePath: filename,
      caption,
      tags: tags ? tags.split(',').map(t => t.trim()).filter(Boolean) : [],
      likesCount: 0,
      commentsCount: 0,
      createdAt: admin.firestore.FieldValue.serverTimestamp()
    };

    await postRef.set(postData);

    // Optionally increment user's post count (transaction)
    const userRef = db.collection('users').doc(uid);
    await db.runTransaction(async tx => {
      const userSnap = await tx.get(userRef);
      if (!userSnap.exists) {
        // create minimal user doc if missing
        tx.set(userRef, { uid, postsCount: 1 }, { merge: true });
      } else {
        const prev = userSnap.get('postsCount') || 0;
        tx.update(userRef, { postsCount: prev + 1 });
      }
    });

    return res.json({ success: true, post: postData });
  } catch (err) {
    console.error('createPost error:', err);
    return res.status(500).json({ error: err.message || 'Server error' });
  }
});

// --------- Quick get post (public) ----------
app.get('/api/post/:id', async (req, res) => {
  try {
    const postId = req.params.id;
    const snap = await db.collection('posts').doc(postId).get();
    if (!snap.exists) return res.status(404).json({ error: 'Not found' });
    return res.json({ post: snap.data() });
  } catch (err) {
    console.error(err);
    return res.status(500).json({ error: 'Server error' });
  }
});

const PORT = process.env.PORT || 3000;
app.listen(PORT, () => console.log(`Server running on port ${PORT}`));

5) Important security & production notes
	1.	Always verify ID tokens on the server using admin.auth().verifyIdToken. The verifyFirebaseIdToken middleware above does that. Do not accept uid from the client without verification.
	2.	Service account JSON must stay secret. Use Replit secrets. Never commit to GitHub.
	3.	Storage public URLs: I used a public URL pattern for simplicity. Prefer signed URLs or restrict Storage access and serve media through your server or via security rules.
	4.	File size limits: multer memory storage uses memory — consider streaming and resumable uploads for large video files or use Firebase client SDK upload directly from the client to Storage (recommended) and only notify server to create post doc. That scales better.
	5.	Content moderation: Add a cloud function or a moderation queue before making posts visible. For MVP you can allow public posts but plan moderation.
	6.	Firestore indexes: If you query for feeds (order by createdAt), create indexes in Firestore console if needed.
	7.	Throttling & rate limits: Implement rate limiting if you expect automated mass requests.

6) Alternative recommended approach (scalable, simpler for clients)

A common pattern for media-first apps:
	•	Client uploads media directly to Firebase Storage using the Firebase client SDK (authenticated user).
	•	After upload completes, client calls your server endpoint or a direct Firestore posts write with the mediaPath and metadata.
	•	Server-side Cloud Function triggers on new posts docs to generate thumbnails, run moderation, send notifications, and update aggregates.

This reduces server memory usage and is faster.
