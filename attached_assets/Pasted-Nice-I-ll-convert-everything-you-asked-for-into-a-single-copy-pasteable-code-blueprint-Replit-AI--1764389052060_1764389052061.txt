4) Cloud functions & Express admin API (cloudFunctions.js)

This file exposes admin endpoints for:
	•	createUser
	•	createPost (server-seeded or moderation)
	•	assignBadge
	•	getBadgeList
	•	seed sample data
It also implements onLike/onComment as Firestore triggers (simulated here as functions — if deploying to Cloud Functions you’d structure differently).

// src/server/cloudFunctions.js
import express from "express";
import bodyParser from "body-parser";
import cors from "cors";
import { auth, db, messaging } from "./adminService.js";
import badges from "./badges.json" assert { type: "json" };

const app = express();
app.use(cors());
app.use(bodyParser.json());

// --- Create user (admin-side) ---
app.post("/admin/createUser", async (req, res) => {
  try {
    const { email, password, displayName, username } = req.body;
    if (!email || !password || !username) return res.status(400).send("missing");
    // Create Firebase Auth user
    const userRecord = await auth.createUser({
      email,
      password,
      displayName
    });
    // Create Firestore profile
    await db.collection("users").doc(userRecord.uid).set({
      uid: userRecord.uid,
      email,
      displayName: displayName || "",
      username,
      avatar: null,
      bio: "",
      followersCount: 0,
      followingCount: 0,
      createdAt: admin.firestore.FieldValue.serverTimestamp(),
      badges: []
    });
    res.json({ uid: userRecord.uid });
  } catch (err) {
    console.error(err);
    res.status(500).send(err.message);
  }
});

// --- Create post (admin or seeded) ---
app.post("/admin/createPost", async (req, res) => {
  try {
    const { uid, mediaUrl, caption, type="image", tags=[] } = req.body;
    if (!uid || !mediaUrl) return res.status(400).send("missing");
    const postRef = db.collection("posts").doc();
    const postData = {
      postId: postRef.id,
      uid,
      type,
      mediaUrl,
      caption: caption || "",
      tags,
      likesCount: 0,
      commentsCount: 0,
      createdAt: admin.firestore.FieldValue.serverTimestamp()
    };
    await postRef.set(postData);
    res.json({ postId: postRef.id });
  } catch (err) {
    console.error(err);
    res.status(500).send(err.message);
  }
});

// --- Assign badge to user ---
app.post("/admin/assignBadge", async (req, res) => {
  try {
    const { uid, badgeId } = req.body;
    if (!uid || !badgeId) return res.status(400).send("missing");
    const userRef = db.collection("users").doc(uid);
    await db.runTransaction(async (tx) => {
      const userSnap = await tx.get(userRef);
      if (!userSnap.exists) throw new Error("user not found");
      const user = userSnap.data();
      const newBadges = (user.badges || []);
      if (!newBadges.includes(badgeId)) newBadges.push(badgeId);
      tx.update(userRef, { badges: newBadges });
    });
    res.json({ ok: true });
  } catch (err) {
    console.error(err);
    res.status(500).send(err.message);
  }
});

// --- Get badge list ---
app.get("/badges", (req, res) => {
  res.json(badges);
});

// --- Realtime-on-like (function to be deployed as Firestore trigger) ---
// For Replit local server we provide function; when deploying to Cloud Functions use exports.onLike = functions.firestore.document('likes/{likeId}')...
export async function handleLike(likeDoc) {
  // likeDoc contains postId and uid
  const { postId } = likeDoc;
  const postRef = db.collection("posts").doc(postId);
  await db.runTransaction(async (tx) => {
    const snap = await tx.get(postRef);
    if (!snap.exists) throw new Error("post missing");
    const prev = snap.data().likesCount || 0;
    tx.update(postRef, { likesCount: prev + 1 });
  });
  // Optionally send notification to post owner
  const postSnap = await postRef.get();
  const post = postSnap.data();
  const ownerId = post.uid;
  const ownerTokenSnap = await db.collection("fcmTokens").doc(ownerId).get();
  if (ownerTokenSnap.exists) {
    const token = ownerTokenSnap.data().token;
    await messaging.send({
      token,
      notification: {
        title: "New Like",
        body: "Someone liked your post!"
      }
    });
  }
}

// --- Start Express server for Replit testing ---
const PORT = process.env.PORT || 3000;
app.listen(PORT, () => console.log("Admin API running on port", PORT));

When deploying to Firebase Cloud Functions, convert endpoints to exports.api = functions.https.onRequest(app) and move handleLike into Firestore triggers using functions.firestore.document('likes/{id}').onCreate(...).

6) Firestore rules (firestore.rules)

This protects data and enforces username format (no emoji: enforce allowed characters).

// firestore.rules
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // Users collection
    match /users/{userId} {
      allow read: if true;
      allow create: if request.auth != null && request.auth.uid == userId
        && isValidUsername(request.resource.data.username);
      allow update: if request.auth != null && request.auth.uid == userId;
      allow delete: if false; // only admin functions handle delete via backend
    }

    // Posts
    match /posts/{postId} {
      allow read: if true;
      allow create: if request.auth != null && request.resource.data.uid == request.auth.uid
        && request.resource.data.mediaUrl is string;
      allow update: if request.auth != null && request.auth.uid == resource.data.uid;
      allow delete: if request.auth != null && request.auth.uid == resource.data.uid;
    }

    // Likes collection: store per-user likes
    match /likes/{likeId} {
      allow create: if request.auth != null && request.resource.data.uid == request.auth.uid;
      allow delete: if request.auth != null && resource.data.uid == request.auth.uid;
      allow read: if request.auth != null;
    }

    // Comments
    match /comments/{commentId} {
      allow create: if request.auth != null && request.resource.data.uid == request.auth.uid;
      allow update, delete: if request.auth != null && request.auth.uid == resource.data.uid;
      allow read: if true;
    }

    // FCM tokens (user device tokens)
    match /fcmTokens/{userId} {
      allow create, update: if request.auth != null && request.auth.uid == userId;
      allow read: if false;
    }

    // Helper for username validation (no emojis)
    function isValidUsername(name) {
      // Allowed: letters, numbers, underscore, dot, dash. 3-24 chars.
      return name is string
        && name.size() >= 3
        && name.size() <= 24
        && name.matches('^[A-Za-z0-9._-]+$');
    }
  }
}

Note: Firestore security rules language does not allow arbitrary function definitions outside match blocks — the above demonstrates intent. When deploying, adapt to exact syntax required (the isValidUsername can be inline or use request.resource.data.username.matches(...)).

// src/server/replit_bot.js
import fetch from "node-fetch";
import { auth, db } from "./adminService.js";

async function createUser(email, password, displayName, username) {
  const userRecord = await auth.createUser({email, password, displayName});
  await db.collection("users").doc(userRecord.uid).set({
    uid: userRecord.uid,
    email,
    displayName,
    username,
    avatar: null,
    bio: "",
    followersCount: 0,
    followingCount: 0,
    createdAt: new Date()
  });
  console.log("Created user", userRecord.uid);
  return userRecord.uid;
}

async function createPost(uid, mediaUrl, caption, tags=[]) {
  const postRef = db.collection("posts").doc();
  await postRef.set({
    postId: postRef.id,
    uid,
    mediaUrl,
    type: mediaUrl.endsWith(".mp4") ? "video" : "image",
    caption,
    tags,
    likesCount: 0,
    commentsCount: 0,
    createdAt: new Date()
  });
  console.log("Created post", postRef.id);
  return postRef.id;
}

// Very small i18n example (JS/React-agnostic)
export const LANGS = {
  en: {
    home: "Home",
    post: "Post",
    likes: "Likes",
    comments: "Comments",
    follow: "Follow",
    // ... more keys
  },

    post: "Firanṣẹ",
    likes: "Gbadun",
    comments: "Awọn asọye",
    follow: "Tẹle",
  }
};

export function setLanguage(langCode) {
  if (!LANGS[langCode]) return false;
  localStorage.setItem("AFRO_LANG", langCode);
  // In your React app, update context/state and re-render all components
  // Example: i18nContext.set(LANGS[langCode])
  return true;
}

export function t(key) {
  const lang = localStorage.getItem("AFRO_LANG") || "en";
  return LANGS[lang][key] || LANGS["en"][key] || key;
}

Frontend usage: import t and call t("home") in UI render. When language changes, update top-level state (React Context) to force re-render so all components use the selected language.

10) Fixes for likes/comments counts (use transactions)

When client writes a likes doc, use a Cloud Function trigger or run a transaction to increment the post’s likesCount reliably.

Example Cloud Function trigger (Firestore onCreate for likes):

// Pseudocode: to deploy as Cloud Function
exports.onLike = functions.firestore.document('likes/{likeId}').onCreate(async (snap, context) => {
  const data = snap.data();
  const postId = data.postId;
  const postRef = admin.firestore().doc(`posts/${postId}`);
  await admin.firestore().runTransaction(async tx => {
    const post = await tx.get(postRef);
    if (!post.exists) return;
    const prev = post.data().likesCount || 0;
    tx.update(postRef, { likesCount: prev + 1 });
  });
});

Same for onDelete of likes (decrement), comments, and comment likes.
