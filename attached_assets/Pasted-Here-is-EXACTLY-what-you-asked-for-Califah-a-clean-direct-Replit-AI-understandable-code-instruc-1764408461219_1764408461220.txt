3. Create a universal bot notification function

// notificationBot.js
import { messaging, db } from "./firebaseAdmin.js";

export const sendBotNotification = async (userId, title, body, type) => {
  const userDoc = await db.collection("users").doc(userId).get();
  const fcmToken = userDoc.data()?.fcmToken;

  if (!fcmToken) return;

  const payload = {
    notification: { title, body },
    data: {
      type,          // e.g. "new_like", "new_comment", "new_follower"
      userId,
    },
    token: fcmToken,
  };

  await messaging.send(payload);

  // Save inside app Notification Center
  await db.collection("users")
    .doc(userId)
    .collection("notifications")
    .add({
      title,
      body,
      type,
      timestamp: Date.now(),
      read: false,
    });
};

A. When someone likes a post

// likePost.js
import { sendBotNotification } from "./notificationBot.js";

export const likePost = async (postId, likerId) => {
  const postRef = db.collection("posts").doc(postId);
  const postDoc = await postRef.get();

  const ownerId = postDoc.data().ownerId;

  // update database like count
  await postRef.collection("likes").doc(likerId).set({
    userId: likerId,
    timestamp: Date.now(),
  });

  // send bot notification
  if (ownerId !== likerId) {
    await sendBotNotification(
      ownerId,
      "New Like ❤️",
      "Someone just liked your post.",
      "new_like"
    );
  }
};

// commentPost.js
import { sendBotNotification } from "./notificationBot.js";

export const commentPost = async (postId, commentId, commenterId, text) => {
  const postRef = db.collection("posts").doc(postId);
  const postDoc = await postRef.get();
  const ownerId = postDoc.data().ownerId;

  // save comment
  await postRef.collection("comments").doc(commentId).set({
    id: commentId,
    userId: commenterId,
    text,
    timestamp: Date.now(),
  });

  // send notification
  if (ownerId !== commenterId) {
    await sendBotNotification(
      ownerId,
      "New Comment 💬",
      "Someone commented on your post.",
      "new_comment"
    );
  }
};

5. Inside the frontend: Handle + display notifications

// notifications.js
import { onSnapshot } from "firebase/firestore";

export const listenToNotifications = (userId, setNotifications) => {
  const ref = collection(db, "users", userId, "notifications");
  onSnapshot(ref, (snapshot) => {
    const list = [];
    snapshot.forEach((doc) => list.push({ id: doc.id, ...doc.data() }));
    setNotifications(list.sort((a, b) => b.timestamp - a.timestamp));
  });
};
